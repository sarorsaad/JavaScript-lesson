app.js
function Header() {
    return (
        <header>
            <h1>My App Header</h1>
        </header>
    );
}

function MainContent() {
    return (
        <main>
            <h2>Main Content</h2>
            <ChildComponent />
        </main>
    );
}

function Footer() {
    return (
        <footer>
            <p>My Footer &copy; 2024</p>
        </footer>
    );
}

function ChildComponent() {
    return (
        <div>
            <p>This is a child component!</p>
        </div>
    );
}

function App() {
    return (
        <div>
            <Header />
            <MainContent />
            <Footer />
        </div>
    );
}

export default App;
