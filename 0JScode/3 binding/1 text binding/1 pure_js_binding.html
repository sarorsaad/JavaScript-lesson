ربط البيانات (Data Binding) في جافا سكريبت النقي

1. ربط النصوص البسيط

هذا المثال يوضح كيفية ربط قيمة متغير مع عنصر HTML:

مرحباً، زائر!

/* ====== View Layer (Presentation) ====== */
// المسؤول عن عرض واجهة المستخدم وتفاعلاتها

// HTML - عناصر واجهة المستخدم
<input type="text" id="nameInput">
<p>مرحباً، <span id="nameOutput">زائر</span>!</p>

// JavaScript - ربط عناصر DOM
const nameInput = document.getElementById('nameInput');
const nameOutput = document.getElementById('nameOutput');


/* ====== Logic Layer (Business Logic) ====== */
// المسؤول عن معالجة البيانات والمنطق البرمجي

// معالجة الأحداث وتحديث البيانات
nameInput.addEventListener('input', function() {
    // منطق معالجة البيانات
    const newValue = this.value || 'زائر';
    
    // تحديث حالة التطبيق
    nameOutput.textContent = newValue;
});

2. ربط خصائص العناصر

هذا المثال يوضح كيفية ربط خصائص العناصر مثل الصور والروابط:

/* ====== View Layer (Presentation) ====== */
// المسؤول عن عرض واجهة المستخدم وتفاعلاتها

// HTML - عناصر واجهة المستخدم
<input type="text" id="imageUrlInput">
<img id="previewImage" src="https://via.placeholder.com/150">

// JavaScript - ربط عناصر DOM
const imageUrlInput = document.getElementById('imageUrlInput');
const previewImage = document.getElementById('previewImage');


/* ====== Logic Layer (Business Logic) ====== */
// المسؤول عن معالجة البيانات والمنطق البرمجي

// معالجة الأحداث وتحديث البيانات
imageUrlInput.addEventListener('input', function() {
    // منطق معالجة البيانات
    const newImageUrl = this.value || 'https://via.placeholder.com/150';
    
    // تحديث حالة التطبيق
    previewImage.src = newImageUrl;
});

3. ربط القوائم

هذا المثال يوضح كيفية ربط مصفوفة مع قائمة:

    /* ====== View Layer (Presentation) ====== */
// المسؤول عن عرض واجهة المستخدم وتفاعلاتها

// HTML - عناصر واجهة المستخدم
<input type="text" id="itemInput">
<button onclick="addItem()">إضافة</button>
<ul id="itemsList"></ul>

// JavaScript - ربط عناصر DOM
let items = ['عنصر 1', 'عنصر 2', 'عنصر 3'];
const itemsList = document.getElementById('itemsList');
const itemInput = document.getElementById('itemInput');


/* ====== Logic Layer (Business Logic) ====== */
// المسؤول عن معالجة البيانات والمنطق البرمجي

// معالجة الأحداث وتحديث البيانات
function renderItems() {
    itemsList.innerHTML = '';
    items.forEach((item, index) => {
        const li = document.createElement('li');
        li.textContent = item;
        const deleteButton = document.createElement('button');
        deleteButton.textContent = 'حذف';
        deleteButton.onclick = () => {
            items.splice(index, 1);
            renderItems();
        };
        li.appendChild(deleteButton);
        itemsList.appendChild(li);
    });
}

function addItem() {
    const newItem = itemInput.value.trim();
    if (newItem) {
        items.push(newItem);
        itemInput.value = '';
        renderItems();
    }
}
