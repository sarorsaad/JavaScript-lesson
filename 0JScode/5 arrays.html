JavaScript Arrays Guide

A Comprehensive Guide to JavaScript Arrays and Array Methods

Array Basics

An array is a special variable that can hold more than one value at a time. It's an ordered collection of data (elements) of any type.

Array Creation:

// Array literal syntax
let fruits = ['apple', 'banana', 'orange'];

// Array constructor syntax
let numbers = new Array(1, 2, 3, 4, 5);

Accessing Elements:

let fruits = ['apple', 'banana', 'orange'];

// Access by index
console.log(fruits[0]);     // 'apple'
console.log(fruits[1]);     // 'banana'

// Get array length
console.log(fruits.length); // 3

Basic Array Methods

push() & pop()

Add/remove elements at the end of array

let arr = [1, 2];
arr.push(3);     // Returns 3 (new length)
console.log(arr); // [1, 2, 3]

arr.pop();       // Returns 3 (removed element)
console.log(arr); // [1, 2]

unshift() & shift()

Add/remove elements at the beginning of array

let arr = [1, 2];
arr.unshift(0);  // Returns 3 (new length)
console.log(arr); // [0, 1, 2]

arr.shift();     // Returns 0 (removed element)
console.log(arr); // [1, 2]

Transformation Methods

map()

Creates new array with results of calling function for every array element

let numbers = [1, 2, 3];
let doubled = numbers.map(num => num * 2);
console.log(doubled); // [2, 4, 6]

filter()

Creates new array with elements that pass test

let numbers = [1, 2, 3, 4, 5];
let evenNumbers = numbers.filter(num => num % 2 === 0);
console.log(evenNumbers); // [2, 4]

reduce()

Reduces array to single value

let numbers = [1, 2, 3, 4];
let sum = numbers.reduce((acc, curr) => acc + curr, 0);
console.log(sum); // 10

Search and Sort Methods

find() & findIndex()

let numbers = [1, 2, 3, 4];
let found = numbers.find(num => num > 2);
console.log(found); // 3

let index = numbers.findIndex(num => num > 2);
console.log(index); // 2

sort()

// String sort
let fruits = ['banana', 'apple', 'orange'];
fruits.sort();
console.log(fruits); // ['apple', 'banana', 'orange']

// Numeric sort
let numbers = [4, 2, 1, 3];
numbers.sort((a, b) => a - b);
console.log(numbers); // [1, 2, 3, 4]

Other Useful Methods

slice() & splice()

// slice()
let fruits = ['apple', 'banana', 'orange'];
let sliced = fruits.slice(1, 2);
console.log(sliced); // ['banana']

// splice()
fruits.splice(1, 1, 'grape');
console.log(fruits); // ['apple', 'grape', 'orange']

concat() & join()

// concat()
let arr1 = [1, 2];
let arr2 = [3, 4];
let combined = arr1.concat(arr2);
console.log(combined); // [1, 2, 3, 4]

// join()
let fruits = ['apple', 'banana'];
let str = fruits.join(', ');
console.log(str); // 'apple, banana'

ES6+ Methods

Array.from() & Array.isArray()

// Array.from()
let arrayFromString = Array.from('hello');
console.log(arrayFromString); // ['h', 'e', 'l', 'l', 'o']

// Array.isArray()
console.log(Array.isArray([1, 2, 3])); // true
console.log(Array.isArray('hello'));   // false

Loop Methods

forEach()

let numbers = [1, 2, 3];
numbers.forEach(num => console.log(num));
// Output:
// 1
// 2
// 3

every() & some()

// every()
let numbers = [2, 4, 6];
console.log(numbers.every(num => num % 2 === 0)); // true

// some()
let nums = [1, 2, 3];
console.log(nums.some(num => num > 2)); // true
